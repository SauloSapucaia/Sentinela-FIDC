// 1. ENSEMBLE DE MACHINE LEARNING (IF + LOF)

# RobustScaler + PCA (15 componentes, preserva variância)
scaler = RobustScaler()
X_scaled = scaler.fit_transform(X_features)
pca = PCA(n_components=15, random_state=42)
X_pca = pca.fit_transform(X_scaled)

# Contamination rate ADAPTATIVO baseado nos sinais da EDA
contamination_rate = max(0.008, min(0.025, sinais_fraude * 1.8))  # → 0.015

# Modelo 1: Isolation Forest (anomalias globais em florestas aleatórias)
modelo_if = IsolationForest(n_estimators=300, contamination=contamination_rate, random_state=42)
modelo_if.fit(X_pca)

# Modelo 2: Local Outlier Factor (anomalias por densidade local)
modelo_lof = LocalOutlierFactor(n_neighbors=20, contamination=contamination_rate)
lof_scores = modelo_lof.fit_predict(X_pca)

# ENSEMBLE: fraude confirmada se capturada por QUALQUER vetor matemático
df['flag_fraude_ia'] = np.where(
    (modelo_if.predict(X_pca) == -1) | (lof_scores == -1), 1, 0
)
# Resultado: 107 anomalias (IF) | ensemble final = 107 boletos

// 2. EXPLAINABLE AI (XAI — Percentis de Referência)

_refs = {col: {'p95': df[col].quantile(0.95), 'p99': df[col].quantile(0.99)}
         for col in _features_explic}

def _explicar(linha):
    motivos = []
    if linha['vlr_nominal'] >= _refs['vlr_nominal']['p99']:
        motivos.append("vlr_nominal extremamente alto")
    if linha['desvio_ticket_pagador'] >= _refs['desvio_ticket_pagador']['p95']:
        motivos.append("foge do padrão histórico do pagador")
    if linha['freq_dupla_pag_benef'] >= _refs['freq_dupla_pag_benef']['p99']:
        motivos.append("repetição anormal entre estas duas empresas")
    return " | ".join(motivos) if motivos else "combinação multivariada fora do padrão"

df['motivo_anomalia_ia'] = np.where(df['flag_fraude_ia'] == 1,
                                     df.apply(_explicar, axis=1), 'não aplicável')

// 3. CONTEXTO MACROECONÔMICO (API BACEN) + Z-SCORE SETORIAL

# BACEN Open Data → Selic atual
selic_atual = 14.75  # % a.a. → fator_risco_macro = +5 pts (threshold > 10%)
df['pontuacao_risco'] += fator_risco_macro  # todos os boletos pré-carregados com +5

# Z-Score setorial (CNAE do beneficiário via API IBGE)
cnae_stats = df.groupby('cd_cnae_prin_beneficiario')['vlr_nominal']\
               .agg(['mean','std']).rename(columns={...})
df['z_score_setorial'] = (df['vlr_nominal'] - df['vlr_medio_cnae']) / df['vlr_std_cnae']
df['flag_anomalia_setorial'] = np.where(df['z_score_setorial'] > 3, 1, 0)
